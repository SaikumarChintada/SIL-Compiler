decl
	Integer fact(Integer n); 
enddecl


Integer fact(Integer n){
decl	
	Integer temp,rval;
enddecl
begin
	
	if (n <= 1) then
		rval = 1;
	else  
		rval = n*fact(n-1);
	endif;
	return rval;
end
}
